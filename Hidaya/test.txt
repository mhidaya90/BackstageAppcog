import { createBackendModule } from '@backstage/backend-plugin-api';
import { createRouter } from './service/router';

export default createBackendModule({
  pluginId: 'my-plugin',
  register(env) {
    env.registerInit({
      deps: {},
      async init() {
        const router = await createRouter();
        env.useRouter('/my-plugin', router);
      },
    });
  },
});
