
// packages/backend/src/plugins/proxy.ts
import { createProxyMiddleware } from 'http-proxy-middleware';

app.use(
  '/bitrise',
  createProxyMiddleware({
    target: 'https://api.bitrise.io/v0.1',
    changeOrigin: true,
    pathRewrite: { '^/bitrise': '' },
    headers: {
      Authorization: process.env.REACT_APP_BITRISE_AUTH_TOKEN!,
    },
  }),
);
