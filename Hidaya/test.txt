import { Router } from 'express';
import { PluginEnvironment } from '../types';
import { bitriseService } from './service';

export async function createRouter(env: PluginEnvironment): Promise<Router> {
  const router = Router();

  router.get('/builds/:appSlug', async (req, res) => {
    const { appSlug } = req.params;
    const service = new bitriseService('your-api-token');
    try {
      const builds = await service.getBuild(appSlug);
      res.json(builds);
    } catch (e) {
      env.logger.error(e);
      res.status(500).json({ error: e.message });
    }
  });

  return router;
}
